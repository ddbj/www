    <li>ultra/CON DF076130-DF076465 (336 エントリ) <br>    Oryzias_latipes_CON_060531_1.seq.gz</li>
    <li>chromosome/CON DG000001-DG000024 (24 エントリ) <br>     Oryzias_latipes_CON_060605_1.seq.gz </li>
</ul>
