<dd>

        <li><i>Prochlorococcus marinus MIT 9515</i> (by J Craig Venter Institute)</li>
        <li><i>Prochlorococcus marinus NATL1A</i> (by J Craig Venter Institute)</li>
        <li><i>Prochlorococcus marinus MIT 9303</i> (by J Craig Venter Institute)</li>
        <li><i>Hyperthermus butylicus DSM 5456</i> (by University of Copenhagen)</li>
    </ul>
